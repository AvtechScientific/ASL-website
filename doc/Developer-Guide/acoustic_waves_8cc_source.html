135 asl::SPDistanceFunction generatePipe(asl::Block & block, Parameters &params)
136 {
137  asl::SPDistanceFunction pipeGeometry;
138  asl::AVec<double> orientation(asl::makeAVec(1., 0., 0.));
139  asl::AVec<double> lVec(asl::makeAVec(params.tubeL.v()+2.*params.dx.v(), 0., 0.));
140  asl::AVec<double> h1Orientation(asl::makeAVec(0., 1., 0.));
141  asl::AVec<double> h2Orientation(asl::makeAVec(0., 0., 1.));
142  asl::AVec<double> center(asl::AVec<double>(params.size)*.5*params.dx.v());
143  double wallMid((params.tubeDEx.v()+params.tubeDIn.v())*.25);
144  double wallTh((params.tubeDEx.v())*.5);
145  asl::AVec<double> h1Center(center - (center*orientation)*orientation +
146  params.hole1Pos.v()*orientation +
147  h1Orientation*wallMid);
148  asl::AVec<double> h2Center(center - (center*orientation)*orientation +
149  params.hole2Pos.v()*orientation +
150  h2Orientation*wallMid);
151 
152  pipeGeometry = asl::generateDFCylinder(params.tubeDEx.v() / 2., lVec, center) &
153  (-asl::generateDFCylinderInf(params.tubeDIn.v() / 2., orientation, center));
154  pipeGeometry = pipeGeometry &
155  (-asl::generateDFCylinder(params.hole1D.v() / 2., h1Orientation * wallTh, h1Center));
156  pipeGeometry = pipeGeometry &
157  (-asl::generateDFCylinder(params.hole2D.v() / 2., h2Orientation * wallTh, h2Center));
158  return asl::normalize(-pipeGeometry, params.dx.v());
159 }
