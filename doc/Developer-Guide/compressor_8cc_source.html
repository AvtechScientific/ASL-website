52 asl::SPDistanceFunction generateCase(asl::Block & bl)
53 {
54  double rCase((bl.getBPosition()[1] - bl.position[1]) / 2.);
55 
56  asl::AVec<> center(.5*(bl.getBPosition() + bl.position));
57  center[1] = bl.position[1] + rCase;
58 
59  auto comprCase(-(generateDFCylinderInf(rCase, makeAVec(0., 0., 1.), center)));
60 
61  return normalize(comprCase, bl.dx);
62 }
