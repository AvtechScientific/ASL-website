118  inline AVec<int> castTransformVector(AVec<int> s)
119  {
120  unsigned int n(s.getSize());
121  AVec<int> r(n, 1);
122  int a(1);
123  for (unsigned int i = 0; i < n - 1; ++i)
124  {
125  a *= s[n - 1 - i];
126  r[n - 2 - i] = a;
127  }
128  return r;
129  }
